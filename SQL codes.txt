----Usecase4: Dailysales Aggregation for same pdt, same store-----------------------------

select Date, p.ProductName, st.StoreName, sum(Quantity) as Totalqty 
from Sales s
join Product p on s.ProductID=p.ProductID
join Store st on s.StoreID=st.StoreID 
group by Date, p. ProductName, st. StoreName
order by  Totalqty desc


----Usecase 5: Monthlysales Aggregation for same pdt, same store -----------------------------

SELECT
     Datepart(Year,Date) as Year,
        DATENAME(MONTH, s.Date) AS Month,
        p.ProductName,
        st.StoreName,
        SUM(Quantity) AS TotalQty
    FROM Sales s
    JOIN Product p
        ON s.ProductID = p.ProductID
    JOIN Store st
        ON s.StoreID = st.StoreID
    GROUP BY
       Datepart(Year,Date),
        DATENAME(MONTH, s.Date),
        p.ProductName,
        st.StoreName
    order by TotalQty desc


---------Usecase6 : Business KPIs-----------------------------------------------------

    --1. Calculating Average selling price for each product / Creating it as a new column


SELECT
    p.ProductID,
    p.ProductName,
    p.SellingPrice,
    CAST(
        SUM((s.Quantity * p.SellingPrice) * (1 - s.DiscountPercent / 100.0))
        / SUM(s.Quantity)
        AS DECIMAL(10,2)
    ) AS ASP
FROM Sales s
JOIN Product p
    ON s.ProductID = p.ProductID
GROUP BY
    p.ProductID,
    p.ProductName,
    p.SellingPrice;




    Alter Table Sales
    ADD SaleAmount DECIMAL(10,2);



    update Sales 
    set 
        s.SaleAmount= (s.Quantity * p.SellingPrice) * (1 - s.DiscountPercent / 100.0)
    from Sales s
    join Product p on s.ProductID=p.ProductID



    select ProductID,sum(SaleAmount)/sum(Quantity) from Sales
    group by ProductID
    order by ProductID




----2. Contribution of specific category to a total salesAmount.

    
    
    select  p.category, sum(s.SaleAmount) as categorysales,
    round(sum(s.SaleAmount) / (select sum (SaleAmount) from Sales) * 100.00,2) as CategoryContributionPercent
    from Sales s
    join Product p on s.ProductID=p.ProductID
    group by  p.category
 


----3. Contribution of specific product to a total salesAmount.

  select p.productID, p.ProductName, sum (s.SaleAmount) as ProductSales,
